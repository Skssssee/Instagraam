    <script>
        // REPLACE THIS with your actual Vercel URL
        const API_BASE = "https://tapi-e.vercel.app/download";

        async function fetchDownloadLinks() {
            const input = document.getElementById('videoInput');
            const btn = document.getElementById('fetchBtn');
            const loader = document.getElementById('loader');
            const resultCard = document.getElementById('resultCard');
            const formatGrid = document.getElementById('formatGrid');

            if (!input.value) return alert("Please paste a URL");

            // UI Reset
            loader.classList.remove('hidden');
            resultCard.classList.add('hidden');
            btn.disabled = true;
            btn.classList.add('opacity-50');

            try {
                // Correctly encoding the URL for your Vercel API
                const encodedUrl = encodeURIComponent(input.value);
                const response = await fetch(`${API_BASE}/${encodedUrl}`);
                const data = await response.json();

                if (data.success) {
                    // Update Details
                    document.getElementById('thumb').src = data.thumbnail || data.thumb;
                    document.getElementById('title').innerText = data.title;
                    
                    // Show Cache Badge if applicable
                    if(data.cached) document.getElementById('cacheBadge').classList.remove('hidden');

                    formatGrid.innerHTML = '';

                    // Logic for mapping q, f, s, u keys
                    data.formats.forEach(item => {
                        const row = document.createElement('div');
                        row.className = "flex items-center justify-between p-4 rounded-2xl bg-gray-800/50 border border-gray-700 hover:border-red-600 transition group";
                        
                        row.innerHTML = `
                            <div class="flex flex-col">
                                <span class="font-bold text-sm uppercase tracking-tighter">${item.q} • ${item.f}</span>
                                <span class="text-xs text-gray-500">${item.s} MB</span>
                            </div>
                            <a href="${item.u}" target="_blank" rel="noopener noreferrer" 
                               class="bg-gray-700 group-hover:bg-red-600 text-white px-5 py-2 rounded-xl text-xs font-bold transition">
                                <i class="fas fa-download mr-1"></i> DOWNLOAD
                            </a>
                        `;
                        formatGrid.appendChild(row);
                    });

                    resultCard.classList.remove('hidden');
                } else {
                    alert("API Error: " + (data.error || "Failed to fetch video."));
                }
            } catch (err) {
                console.error(err);
                alert("Connection failed. Make sure your Vercel API is online!");
            } finally {
                loader.classList.add('hidden');
                btn.disabled = false;
                btn.classList.remove('opacity-50');
            }
        }
    </script>
